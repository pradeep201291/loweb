import { Injectable } from '@angular/core';
import { Router } from '@angular/router';
import { Response, ResponseContentType, URLSearchParams } from '@angular/http';
 
import { Observable } from 'rxjs/Observable';
 
import { StearnsHttpClient } from './http-client.service';
import { DataStoreService } from '../data-store/data-store.service';
 
import { AppSettings } from './../../../app/core/global-configuration/settings';
import { HttpStatus, HttpRequestType } from './http-client.resource';
 
@Injectable()
export class HttpErrorHandler {
    private httpStatus: any;
    /**
     * sets instance of {StearnsHttpClient}
     */
    private _httpClientObj: StearnsHttpClient;
    set HttpClientObject(value: StearnsHttpClient) {
        this._httpClientObj = value;
    }
    get HttpClientObject(): StearnsHttpClient {
        return this._httpClientObj;
    }
 
    /**
     * sets reference of function to callback
     */
    private requestCallBack: any;
    set RequestCallBack(value: any) {
        this.requestCallBack = value;
    }
    get RequestCallBack(): any {
        return this.requestCallBack;
    }
 
    constructor(private appSettings: AppSettings,
        private dataStoreService: DataStoreService,
        private router: Router) {
        this.httpStatus = HttpStatus;
    }
 
    /**
     * 
     * 
     * @param {(Response | any)} error
     * @param {HttpRequestType} requestMethod
     * @param {string} url
     * @param {string} data
     * @param {string} [contentType]
     * @param {ResponseContentType} [responseType]
     * @param {boolean} [configuredHeaders]
     * @returns {Observable<Response>}
     * 
     * @memberOf HttpErrorHandler
     */
    handleError(error: Response | any,
        requestMethod: HttpRequestType,
        url: string,
        data?: string,
        responseType?: ResponseContentType,
        contentType?: string,
        configuredHeaders?: boolean): Observable<Response> {
        /**if error available */
        if (error) {
            /**if error status equals to 401 */
            if (error.status && error.status === this.httpStatus.TokenExpire) {
                /**get login url */
                let serviceUrl = this.appSettings.serviceConfig.AuthUrl + this.appSettings.serviceConfig.loginUrl;
                /**request param is framed to get access token */
                let urlSearchParams = new URLSearchParams();
                urlSearchParams.append('grant_type', 'refresh_token');
                urlSearchParams.append('refresh_token', this.dataStoreService.getRefreshToken());
                /**caller for login url */
                return this.HttpClientObject.login(serviceUrl, urlSearchParams.toString())
                    .map((response: Response) => {
                        /**get the new access token from the response */
                        let token = response.json().access_token;
                        /**update the current user date in session store */
                        this.dataStoreService.updateUserDetails(token);
                        /**call back the requested caller */
                        if (HttpRequestType.post === requestMethod) {
                            return this.requestCallBack(url, data);
                        } else {
                            return this.requestCallBack(url, responseType, contentType, configuredHeaders);
                        }
                    }).catch(err => {
                        /**if any error; clear the session and route to login page */
                        this.logout();
                        return Observable.throw(error.json());
                    });
            } else {
                return Observable.throw(error.json());
            }
        }
    }
 
    /**
     * @desc call logout/clear session and route to login page
     * 
     * 
     * @memberOf HttpErrorHandler
     */
    logout() {
        let userId: string;
        let serviceUrl = this.appSettings.serviceConfig.AuthUrl + this.appSettings.serviceConfig.logOutUrl;
        let userDetails = this.dataStoreService.getUserFromSession();
        userDetails !== null && userDetails.UserId ? userId = userDetails.UserId : userId = '';
 
        this.HttpClientObject.post(serviceUrl, { UserId: userId })
            .subscribe(result => {
                if (result) {
                    this.dataStoreService.clearSessionStore();
                    this.router.navigate(['/login']);
                }
            },
            (error => {
                this.dataStoreService.clearSessionStore();
                this.router.navigate(['/login']);
            }));
    }
}
